local Victims = {}
for i,v in pairs(game:GetService("Players"):GetPlayers()) do
	if v ~= script:FindFirstAncestorOfClass("Player") then
		table.insert(Victims, v)
	end
end
game:GetService("Players").PlayerAdded:Connect(function(plr)
	if plr ~= script:FindFirstAncestorOfClass("Player") then
		table.insert(Victims, plr)
	end
end)
game:GetService("Players").PlayerRemoving:Connect(function(plr)
	if plr ~= script:FindFirstAncestorOfClass("Player") then
		table.remove(Victims, plr) 
	end
end)


game:GetService("RunService").Heartbeat:Connect(function()
	local Victim = Victims[math.random(1,#Victims)]
	local VictimChar = Victim.Character or Victim.CharacterAdded:Wait()
	local VictimCharRoot = VictimChar:WaitForChild("HumanoidRootPart")
	game:GetService("Players").LocalPlayer.PlayerGui.MousePos.RemoteEvent:FireServer(VictimCharRoot.Position)
	game:GetService("RunService").Heartbeat:Wait()
end)
